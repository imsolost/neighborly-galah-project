function factorialize(num) {
  if (num === 0) {
    return 1;
  }
  return num * factorialize(num - 1);

}


factorialize(5);

function palindrome(str) {
  var mod = str.toLowerCase().replace(/[\s&\/\\#,+()$~%.'":*?<>{}_-]/g, '');
  var reverse = mod.split('').reverse().join('');
  return mod === reverse;
}



palindrome("eye");

find longest word

function findLongestWord(str) {
  // break into array push to accumulator and compare return the longest one
  var splitted = str.split(' ');
  var acc = 0;
  for (var i=0; i < splitted.length; i++){
    // if something greater than accumulator
    if (splitted[i].length > acc){
      acc = splitted[i].length;
    }
  }


  return acc;
}

findLongestWord("The quick brown fox jumped over the lazy dog");

function titleCase(str) {
  var array = str.toLowerCase().split(' ');
  var accum = [];
  for (var i=0; i < array.length; i++){
    var string = array[i];
    accum.push(string[0].toUpperCase() + string.substring(1));
  }
return accum.join(' ');}

function truncateString(str, num) {
  if (num < 4 && str.length > num) {
    return str.slice(0, num) + "...";

  } else if (num > 3 && str.length > num) {
  return str.slice(0, num - 3) + "...";

  } else {
    return str;
  }
}
